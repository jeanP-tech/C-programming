배열의 홀짝 판별
========

```
#include <stdio.h>

int main() {
  
  int arr[10] = { 5, 10, 15, 22, 23, 102, 99, 102, 130, 8 };
  int odd = 0;
  int even = 0;
  
  for(int i = 0; i < 10; i++) {
		if(arr[i] % 2 == 0) {
			even += 1;
		}
		else {
			odd += 1;
		}
	}
  
  printf("홀수의 갯수는 %d개 입니다.\n", odd);
  printf("짝수의 갯수는 %d개 입니다.\n", even);
  
  return 0;
}
```

가장 작은 수
========
```
#include <stdio.h>

int main() {
  
  int arr[10] = { 1032, 99, 1004, 234, 452, 8, 922, 445, 246, 2048 };
  int min = arr[0];
  int i; // 반복문에 쓰일 변수
  
	for (int i = 1; i < 11; i++) {
		if (min > arr[i]) {
			min = arr[i];
		}
		else {
			min = min;
		}
	}
  printf("배열 안에서 가장 작은 수는 %d입니다.\n", min);
  
  return 0;
}
```
